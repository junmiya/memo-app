# ジュニアプログラマー向け メモサイト設計書

## 1. 学習目標

このプロジェクトで学べること：
- **HTML/CSS/JavaScript** の基本的な使い方

## 2. プロジェクト概要

### 2.1 シンプルなメモアプリ
- 1つの `Memo` モデルのみ使用
- 表面・裏面を持つカードデザイン
- ローカルストレージでデータ保存
- バニラJavaScript（フレームワーク不使用）

### 2.2 最小限の機能
- メモの作成・編集・削除
- カードのフリップアニメーション
- シンプルな一覧表示

### 5.3 JavaScript（script.js）

```javascript
// メモアプリのメインクラス
class MemoApp {
    constructor() {
        this.memos = [];
        this.currentEditingId = null;
        this.init();
    }

    // 初期化
    init() {
        this.loadMemos();
        this.bindEvents();
        this.renderMemos();
    }

    // イベントのバインド
    bindEvents() {
        // 新規メモボタン
        document.getElementById('add-memo-btn').addEventListener('click', () => {
            this.showModal();
        });

        // モーダルのキャンセルボタン
        document.getElementById('cancel-btn').addEventListener('click', () => {
            this.hideModal();
        });

        // フォーム送信
        document.getElementById('memo-form').addEventListener('submit', (e) => {
            e.preventDefault();
            this.saveMemo();
        });

        // モーダルの背景クリックで閉じる
        document.getElementById('modal').addEventListener('click', (e) => {
            if (e.target.id === 'modal') {
                this.hideModal();
            }
        });
    }

    // メモの生成
    createMemo(title, content, color) {
        return {
            id: Date.now().toString(), // 簡単なID生成
            title: title,
            content: content,
            color: color,
            createdAt: new Date().toISOString()
        };
    }

    // メモの保存
    saveMemo() {
        const title = document.getElementById('memo-title').value;
        const content = document.getElementById('memo-content').value;
        const color = document.getElementById('memo-color').value;

        if (!title || !content) {
            alert('タイトルと内容を入力してください');
            return;
        }

        if (this.currentEditingId) {
            // 編集モード
            const memo = this.memos.find(m => m.id === this.currentEditingId);
            memo.title = title;
            memo.content = content;
            memo.color = color;
        } else {
            // 新規作成
            const newMemo = this.createMemo(title, content, color);
            this.memos.push(newMemo);
        }

        this.saveMemos();
        this.renderMemos();
        this.hideModal();
    }

    // メモの削除
    deleteMemo(id) {
        if (confirm('本当にこのメモを削除しますか？')) {
            this.memos = this.memos.filter(memo => memo.id !== id);
            this.saveMemos();
            this.renderMemos();
        }
    }

    // メモの編集
    editMemo(id) {
        const memo = this.memos.find(m => m.id === id);
        if (memo) {
            this.currentEditingId = id;
            document.getElementById('modal-title').textContent = 'メモを編集';
            document.getElementById('memo-title').value = memo.title;
            document.getElementById('memo-content').value = memo.content;
            document.getElementById('memo-color').value = memo.color;
            this.showModal();
        }
    }

    // カードのフリップ
    flipCard(cardElement) {
        cardElement.classList.toggle('flipped');
    }

    // メモの描画
    renderMemos() {
        const grid = document.getElementById('memo-grid');
        grid.innerHTML = '';

        this.memos.forEach(memo => {
            const cardElement = this.createMemoCard(memo);
            grid.appendChild(cardElement);
        });
    }

    // メモカードのHTML要素を作成
    createMemoCard(memo) {
        const card = document.createElement('div');
        card.className = `memo-card ${memo.color}`;
        card.innerHTML = `
            <div class="memo-card-inner">
                <div class="memo-card-front">
                    <div class="memo-title">${memo.title}</div>
                    <div class="memo-actions">
                        <button onclick="app.editMemo('${memo.id}')">編集</button>
                        <button onclick="app.deleteMemo('${memo.id}')">削除</button>
                    </div>
                </div>
                <div class="memo-card-back">
                    <div class="memo-content">${memo.content}</div>
                    <div class="memo-actions">
                        <button onclick="app.editMemo('${memo.id}')">編集</button>
                        <button onclick="app.deleteMemo('${memo.id}')">削除</button>
                    </div>
                </div>
            </div>
        `;

        // カードクリックでフリップ
        card.addEventListener('click', (e) => {
            // ボタンクリック時はフリップしない
            if (e.target.tagName === 'BUTTON') {
                return;
            }
            this.flipCard(card);
        });

        return card;
    }

    // モーダル表示
    showModal() {
        this.currentEditingId = null;
        document.getElementById('modal-title').textContent = '新しいメモ';
        document.getElementById('memo-form').reset();
        document.getElementById('modal').classList.remove('hidden');
    }

    // モーダル非表示
    hideModal() {
        document.getElementById('modal').classList.add('hidden');
        this.currentEditingId = null;
    }

    // ローカルストレージからメモを読み込み
    loadMemos() {
        const stored = localStorage.getItem('memos');
        if (stored) {
            this.memos = JSON.parse(stored);
        }
    }

    // ローカルストレージにメモを保存
    saveMemos() {
        localStorage.setItem('memos', JSON.stringify(this.memos));
    }
}

// アプリケーションの開始
const app = new MemoApp();
```

### 6.3 Step 3: ローカルストレージ
```javascript
// データを保存
function saveMemo(memo) {
    const memos = JSON.parse(localStorage.getItem('memos')) || [];
    memos.push(memo);
    localStorage.setItem('memos', JSON.stringify(memos));
}

// データを読み込み
function loadMemos() {
    const stored = localStorage.getItem('memos');
    return stored ? JSON.parse(stored) : [];
}
```
